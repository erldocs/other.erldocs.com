logi_channel

Log Message Channels.

Log Message Channels

A channel manages conditional sinks

  %%%
  %%% Example
  %%%
 
  %%
  %% CREATE
  %%
  > ok = logi_channel:create(sample_log).
  > logi_channel:which_channels().
  [sample_log,logi_default_log]  % 'logi_default_log' is created automatically when 'logi' application was started
 
  %%
  %% INSTALL SINK
  %%
  > Sink = logi_sink:new(logi_builtin_sink_null).
  > {ok, _} = logi_channel:install_sink(sample_log, Sink).
  > logi_channel:which_sinks(sample_log).
  [logi_builtin_sink_null]

Types

id() = atom()

install_sink_option() = {lifetime, timeout() | pid()}
                              | {if_exists, error | ignore | supersede}

install_sink_options() = [install_sink_option()]

install_sink_result() = {ok,
                                 OldSink :: undefined
                                          | logi_sink:sink()}
                              | {error,
                                 {already_installed, logi_sink:sink()}}

uninstall_sink_result() = {ok, logi_sink:sink()} | error

Functions

default_channel() -> id()

The default channel

This channel is created automatically when logi application was started.

create(Channel::id()) -> ok

Creates a new channel

If the channel already exists, nothing happens.

If there exists a process or a ETS table with the same name as Channel, the function crashes.

delete(Channel::id()) -> ok

Deletes a channel

If the channel does not exists, it is silently ignored.

which_channels() -> [id()]

Returns a list of all existing channels

install_sink(Channel::id(), Sink::sink() (see module logi_sink)) -> install_sink_result()

Equivalent to install_sink(Channel, Sink, []).

install_sink(Channel::id(), Sink::sink() (see module logi_sink), Options::install_sink_options()) -> install_sink_result()

Installs Sink into Channel

uninstall_sink(Channel::id(), SinkId::id() (see module logi_sink)) -> uninstall_sink_result()

Uninstalls the sink which has the identifier SinkId from Channel

find_sink(Channel::id(), SinkId::id() (see module logi_sink)) -> {ok, sink() (see module logi_sink)} | error

Searchs for SinkId in Channel; returns {ok, Sink}, or error if SinkId is not present

which_sinks(Channel::id()) -> [id() (see module logi_sink)]

Returns a list of installed sinks

set_condition(Channel::id(), SinkId::id() (see module logi_sink), Condition::condition() (see module logi_sink)) -> {ok, condition() (see module logi_sink)} | error

Sets the condition of the sink which has the identifier SinkId to Condition

This function returns {ok, OldCondition} if specified sink exists, error otherwise. OldCondition is the old condition of the sink.
