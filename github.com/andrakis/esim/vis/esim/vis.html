Simulator visualizer.

Simulator visualizer.

This module provides the base for a visualizer to render the world. It requires an implementation to work with.

Current ideas for visualizers: http interface, console output

This module, the base implementation, implements a text mapping that may be printed to console.

loc() = location_id() | location()

location() =
            #location{id = undefined | location_id(),
                      name = undefined | binary(),
                      parent = undefined | pid(),
                      sub_locations = [pid()],
                      actors = [pid()],
                      neighbours = [neighbour()],
                      iteration_reference = undefined | reference(),
                      iteration_waitlist = [pid()],
                      iteration_callee = undefined | pid(),
                      state = undefined | term(),
                      snapshot = [term()],
                      snapshot_building = [term()],
                      lookup = any()}

location_id() = pid()

neighbour() =
            #neighbour{direction = undefined | direction(),
                       id = undefined | location_id()}

pos() = {X :: integer(), Y :: integer()}

vis() =
            #vis{locations = {location_id(), loc()},
                 root = undefined | loc(),
                 min_x = integer() =< -1,
                 min_y = integer() =< -1,
                 max_x = integer() >= 1,
                 max_y = integer() >= 1,
                 visual = visual_map(),
                 done = [{pos(), done}]}

visual_map() = [{pos(), binary()}]
