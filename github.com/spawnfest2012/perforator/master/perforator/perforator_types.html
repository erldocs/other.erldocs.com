desc_decorator() = {desc, string(), test_fun()}

fixture_test_obj() = setup_fixture() | foreach_fixture()

foreach_fixture() =
            {foreach,
             Setup :: fun(() -> any()),
             Cleanup :: fun((any()) -> any()),
             [test_obj()]}

fun_spec() = mfa_spec() | function()

mfa_spec() =
            {Module :: atom(),
             Funtion :: atom(),
             Arity :: integer() >= 0}

primitive_test_obj() = test_fun()
                             | test_fun_decorator()
                             | function()

repeat_decorator() =
            {repeat,
             test_fun(),
             Times :: integer() >= 1,
             Sleep :: timer:time()}

run_results() = term()

setup_fixture() =
            {setup,
             Setup :: fun(() -> any()),
             Cleanup :: fun((any()) -> any()),
             test_obj()}

test_case_opts() = term()

test_case_results() = term()

test_fun() = {raw_fun, mfa_spec()}

test_fun_decorator() = repeat_decorator() | desc_decorator()

test_obj() = primitive_test_obj() | fixture_test_obj()
