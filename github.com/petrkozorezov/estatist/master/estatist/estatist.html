Fast application statistics agregator for erlang, based on basho_metrics.

Types

app_options() = [{metrics, metrics()}]

functor() = {atom(), atom(), list()} | fun(() -> any())

metric_id() = metric_name() | {metric_name(), row_id()}

metric_input_value() = number()

metric_name() = atom() | string()

metric_scalarity() = tbl | var

metric_type() = atom()

metric_type_options() = any()

metric_type_param() = atom()

metric_types() =
            [metric_type() | {metric_type(), metric_type_options()}]

metric_value() = metric_input_value() | undefined

metrics() =
            [{metric_name(), metric_scalarity(), metric_types()}]

row_id() = atom() | string() | binary() | integer()

select_param(T) = all | T | [T]

select_query() =
            [{names, select_param(metric_name())} |
             {types, select_param(metric_type())} |
             {params, select_param(metric_type_param())} |
             {row_id, select_row_id()}]

select_results() = [select_results_metric()]
                         | select_results_metric()

select_results_metric() =
            select_results_tree(metric_name(), select_results_type())

select_results_param() =
            select_results_tree(metric_type_param(), metric_value())

select_results_tree(K, V) = {K, [V]} | {K, V} | V

select_results_type() =
            select_results_tree(metric_type(), select_results_param())

select_row_id() = all
                        | first
                        | last
                        | {next, row_id()}
                        | {prev, row_id()}
                        | {id, row_id()}
                        | [{id, row_id()}]

Update specific metric with value, which will be applied to all instances of metric types.

tc_update(MetricID::metric_id(), Fun::functor()) -> any()

Same as tc_update(throw, MetricID, Fun)

tc_update(ThrowFlag::throw | quite, MetricID::metric_id(), Fun::functor()) -> any()

Run Fun with timer:tc and then update metric with resutls time. ThrowFlag means, throw or not exception when update fails.

get(MetricID::metric_id(), Type::metric_type(), Param::metric_type_param()) -> metric_value()

Get specific value by: metric name, metric type, type param.

select_all() -> {ok, select_results()} | {error, term()}

Same as select([])

select(QueryPL::select_query()) -> {ok, select_results()} | {error, term()}

Select values from all metrics. You may specify following conditions:
* names -- metrics names;
* types -- metrics types;
* params -- metrics types params;
* row_id -- row identificators, when metric is table.
Run without conditions will retutn all information. See examples in main README.

add_metric(Name::metric_name(), Scalarity::metric_scalarity(), MetricTypes::metric_types()) -> ok | {error, term()}

Add new metric on runtime.

delete_metric(Name::metric_name()) -> ok

Delete metric on runtime.

measure_call_time_simple(Fun::functor()) -> pos_integer()

measure_call_time(Fun::functor()) -> histogram_stats() (see module basho_metrics_nifs)

Same as measure_call_time(Fun, {time, 10000000}) .

measure_call_time(Fun::functor(), Options::{time | count, pos_integer()}) -> histogram_stats() (see module basho_metrics_nifs)

Measuring function executing time. You can set time or count of repeating executions by Options param.
