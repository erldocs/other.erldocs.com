deps() = ordsets:ordset(remote_type())

erl_spec() = [erl_spec_clause(), ...]

erl_spec_clause() = {[erl_type()], erl_type()}

erl_type() = any
                   | atom
                   | {atom, atom()}
                   | boolean
                   | float
                   | integer
                   | {integer, integer()}
                   | {list, erl_type()}
                   | {nonempty_list, erl_type()}
                   | nil
                   | {range, integer() | inf, integer() | inf}
                   | tuple
                   | {tuple, [erl_type()]}
                   | {union, [erl_type()]}

raw_type() = t_any()
                   | t_nil()
                   | t_atom()
                   | t_atom_lit()
                   | t_integer()
                   | t_integer_lit()
                   | t_float()
                   | t_tuple()
                   | t_list()
                   | t_nonempty_list()
                   | t_union()
                   | t_range()
                   | t_bitstring()
                   | t_function()
                   | t_local()

                   | t_record()
                   | t_type_var()

record_field_name() = atom()

record_field_type() = {record_field_name(), raw_type()}

record_name() = atom()

remote_type() = {module(), type_name(), type_arity()}

stored_type_key() = {record, record_name()}
                          | {type, type_name(), type_arity()}

stored_type_value() = [record_field_type()]
                            | {raw_type(), [type_var()]}

stored_types() =
            dict:dict(stored_type_key(), stored_type_value())

t_any() = #t{kind = simt, rep = any, deps = deps()}

t_atom() = #t{kind = simt, rep = atom, deps = deps()}

t_atom_lit() =
            #t{kind = simt, rep = {atom, atom()}, deps = deps()}

t_bitstring() =
            #t{kind = simt, rep = {bitstring, 1 | 8}, deps = deps()}

t_constraint() = {t_type_var(), raw_type()}

t_float() = #t{kind = simt, rep = float, deps = deps()}

t_function() = #t{kind = simt,
                          rep = function,
                          deps = deps()}
                     | t_function_det()

t_function_det() =
            #t{kind = simt,
               rep =
                   {function, [raw_type()], raw_type(), [t_constraint()]},
               deps = deps()}

t_integer() = #t{kind = simt, rep = integer, deps = deps()}

t_integer_inf() = t_integer_pos_inf() | t_integer_neg_inf()

t_integer_lit() =
            #t{kind = simt, rep = {integer, integer()}, deps = deps()}

t_integer_neg_inf() =
            #t{kind = simt, rep = inf, deps = deps()}

t_integer_pos_inf() =
            #t{kind = simt, rep = inf, deps = deps()}

t_list() =
            #t{kind = simt, rep = {list, raw_type()}, deps = deps()}

t_local() =
            #t{kind = loct,
               rep = {local, type_name(), [raw_type()]},
               deps = deps()}

t_nil() = #t{kind = simt, rep = nil, deps = deps()}

t_nonempty_list() =
            #t{kind = simt,
               rep = {nonempty_list, raw_type()},
               deps = deps()}

t_range() =
            #t{kind = simt,
               rep = {range, t_range_limit(), t_range_limit()},
               deps = deps()}

t_range_limit() = t_integer_lit() | t_integer_inf()

t_record() =
            #t{kind = rect,
               rep = {record, record_name(), [record_field_type()]},
               deps = deps()}

t_remote() =
            #t{kind = remt,
               rep = {remote_type, module(), type_name(), [raw_type()]},
               deps = deps()}

t_tuple() = #t{kind = simt, rep = tuple, deps = deps()}
                  | #t{kind = simt,
                       rep = {tuple, [raw_type()]},
                       deps = deps()}

t_type_var() =
            #t{kind = vart, rep = type_var(), deps = deps()}

t_union() =
            #t{kind = simt, rep = {union, [raw_type()]}, deps = deps()}

type_arity() = byte()

type_name() = atom()

type_var() = {tvar, atom()}
