em_telnet

Handle incoming data and option negotiation in a telnet session.

Handle incoming data and option negotiation in a telnet session.

Types

orddict(Key, Val) = [{Key, Val}]

parse_mode() = text | eol | cmd | sub

socket() = port()

telnet_disable_opt() = 252 | 254

telnet_enable_opt() = 251 | 253

telnet_option() = byte()

telnet_options() =
            orddict(telnet_option(),
                    #telopt{current = telopt_state(),
                            queue = telopt_queue()})

telnet_printer_fun() = fun((string()) -> ok)

telopt_queue() = 'EMPTY' | 'OPPOSITE'

telopt_state() = 'NO' | 'YES' | 'WANTNO' | 'WANTYES'
