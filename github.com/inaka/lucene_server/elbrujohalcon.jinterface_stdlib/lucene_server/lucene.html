lucene

Lucene Interface.

Lucene Interface. Use it to add/del/query documents

Types

doc() = [{field_key(), field_value()}, ...]

field_key() = atom() | binary() | string()

field_value() = number() | atom() | string() | geo()

geo() =
            #geo{lat = undefined | float(), lng = undefined | float()}

metadata() =
            [{page_token, page_token()} |
             {total_hits, integer() >= 0} |
             {first_hit, integer() >= 0}]

page_token() = binary()
