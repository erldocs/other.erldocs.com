elogger

The real logging module behaviour.

The real logging module behaviour. Defines how the messages will be logged. The user module should export: 

init(Args::term()) -> init_result()

Opens and/or initializes the client.

log(Message::log(), State::term()) -> {ok, NewState::term()}

Called each time a message is received

handle_info(Message::term(), State::term()) -> {ok, NewState::term()}

Called each time an erlang message is received

terminate(Reason :: normal | shutdown | term(), State) -> _

Let the user module clean up. Always called when server terminates.

Types

exception_class() = mod | cat | re

init_result() = {ok, State :: term()}
                      | ignore
                      | {stop, Reason :: term()}

log() =
            #log{time =
                     {{2010..9999, 1..12, 1..31}, {0..23, 0..59, 0..59}},
                 level = undefined | elog:loglevel(),
                 category = atom(),
                 module = undefined | atom(),
                 line = undefined | integer(),
                 pid = undefined | pid(),
                 node = undefined | node(),
                 stacktrace = [term()],
                 text = string(),
                 args = [term()]}
