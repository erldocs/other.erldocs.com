This is a module for acknowledge to clients. Default acknowledger for gg_framework. The acknowledge action defined in include/config/ack_action.hrl.

Usage: File: include/config/ack_action.hrl 

  ack_action() ->
    AckAction = [
      [100 | [
        {params, [
          {username, string},
          {Age, integer}
        ]}
      ]]
    ],
    {ok. AckAction}.
  AckCode = 100,
