tg_store

Associative store in which values are associated with tags.

Associative store in which values are associated with tags. It is possible to store a tag/value pair, and to look up the value(s) associated with a tag.

Types

empty_store() = []

store() = empty_store() | [{atom(), any()}]

Functions

new() -> {ok, empty_store()}

size(Store::store()) -> non_neg_integer()

add(Tag::atom(), Value::any(), Store::store()) -> {ok, store()}

lookup(Tag::atom(), Store::store()) -> {ok, any()} | {error, not_found}
