agr_type() = count
                   | min
                   | max
                   | sum
                   | avg
                   | quantize
                   | lquantize
                   | llquantize

dtrace_consume_data() = [probe()]

dtrace_error() = {error, {dtrace, string()}}

dtrace_handle() = binary()

dtrace_walk_data() = [walk_rec()]

probe() = {probe, probe_desc()}
                | {probe, probe_desc(), probe_rec()}

probe_desc() = [string()]

probe_rec() = term()

walk_rec() = {agr_type(), probe_rec(), number()}

Functions

script(Script::string()) -> {ok, Handle::dtrace_handle()} | dtrace_error()

open() -> {ok, Handle::dtrace_handle()} | dtrace_error()

setopt(Handle::dtrace_handle(), Option::string(), Value::string()) -> ok | dtrace_error()

compile(Handle::dtrace_handle(), Script::string()) -> ok | {error, already_running} | dtrace_error()

go(Handle::dtrace_handle()) -> ok | {error, no_prog}

stop(Handle::dtrace_handle()) -> ok | {error, no_prog}

consume(Handle::dtrace_handle()) -> ok | {ok, dtrace_consume_data()} | dtrace_error() | {error, no_prog}

walk(Handle::dtrace_handle()) -> ok | {ok, dtrace_walk_data()} | dtrace_error() | {error, exited | filled | stopped | no_prog}
