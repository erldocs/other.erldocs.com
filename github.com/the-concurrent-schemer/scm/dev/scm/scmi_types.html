            #error_read{lineno = undefined | scmd_types_impl:lineno(),

iodev() = scmi_iodev:iodev()

lip() = lip0() | lipn() | lipv() | lipnv()

lip0() =
            #lip0{lineno = undefined | scmd_types_impl:lineno(),
                  val = l0()}

lipn() =
            #lipn{lineno = undefined | scmd_types_impl:lineno(),
                  val = ln() | [ln()]}

lipnv() =
            #lipnv{lineno = undefined | scmd_types_impl:lineno(),
                   val = lnv()}

lipv() =
            #lipv{lineno = undefined | scmd_types_impl:lineno(),
                  val = lv()}

ln() =
            #ln{params = params(),
                body = body(),
                env = denv(),
                src = src()}

lnv() =
            #lnv{n = integer() >= 1,
                 params = params(),
                 body = body(),
                 env = denv(),
                 src = src()}

lv() =
            #lv{param = param(),
                body = body(),
                env = denv(),
                src = src()}

            #nip0{lineno = undefined | scmd_types_impl:lineno(),

nipn() =
            #nipn{lineno = undefined | scmd_types_impl:lineno(),

nipnv() =
            #nipnv{lineno = undefined | scmd_types_impl:lineno(),
                   val = fnv()}

nipv() =
            #nipv{lineno = undefined | scmd_types_impl:lineno(),
                  val = fv()}
