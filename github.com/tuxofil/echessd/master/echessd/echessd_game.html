                    | {status, status()}
                    | {winner, echessd_user:name()}
                    | {winner_color, color()}
                    | {draw_request_from, echessd_user:name()}
                    | {acknowledged, boolean()}

ply() = {ply_coords(), ply_info()}

ply_coords() = nonempty_string()

ply_info() = [ply_info_item()]

ply_info_item() = {time, GregorianSeconds :: integer() >= 0}

                           ChessNotation :: nonempty_string()}

status() = alive | checkmate | draw_stalemate
