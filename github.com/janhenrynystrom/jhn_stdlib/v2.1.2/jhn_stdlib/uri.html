Types

opt() = none

scheme() = http | https | file

uri() =
            #uri{scheme = undefined | scheme(),
                 userinfo = [binary()],
                 host = undefined | binary() | inet:ip_address(),
                 port = undefined | inet:port_number(),
                 path = [binary()],
                 query = binary(),
                 fragment = binary()}

encode(Term::uri()) -> iolist()

Encodes the structured Erlang term as an iolist. Equivalent of encode(Term, []) -> URI.

encode(Term::uri(), Opts::[opt()] | #opts{ipv4=boolean(), return_type=iolist | binary, orig_call=any()}) -> iolist() | binary()

Encodes the structured Erlang term as an iolist or binary. Encode will give an exception if the erlang term is not well formed. Options are: binary -> a binary is returned iolist -> a iolist is returned ipv4 -> encoded IPv6 host address has the two least sigificant segments repesented in IPv4 address format
