msgpack_packer

Types

format_type() = jsx | jiffy | map

msgpack_ext_packer() =
            fun((tuple(), msgpack:options()) ->
                    {ok, {Type :: byte(), Data :: binary()}} |
                    {error, any()})

msgpack_ext_unpacker() = fun((byte(),
                                      binary(),
                                      msgpack:options()) ->
                                         {ok, msgpack_term()} |
                                         {error, any()})
                               | fun((byte(), binary()) ->
                                         {ok, msgpack_term()} |
                                         {error, any()})

msgpack_list_options() =
            [{format, format_type()} |
             jsx |
             jiffy |
             {allow_atom, none | pack} |
             {enable_str, boolean()} |
             {ext,
              {msgpack_ext_packer(), msgpack_ext_unpacker()} | module()}]

msgpack_map() = msgpack_map_jsx() | msgpack_map_jiffy() | #{}

msgpack_map_jiffy() = {[{msgpack_term(), msgpack_term()}]}

msgpack_map_jsx() = [{msgpack_term(), msgpack_term()}] | [{}]

msgpack_map_unpacker() =
            fun((binary(), integer() >= 0, msgpack_option()) ->
                    {msgpack_map(), binary()} | no_return())

msgpack_option() =
            #options_v3{interface = format_type(),
                        map_unpack_fun = msgpack_map_unpacker(),
                        impl = erlang | nif,
                        allow_atom = none | pack,
                        enable_str = boolean(),
                        ext_packer = msgpack_ext_packer() | undefined,
                        ext_unpacker = msgpack_ext_unpacker() | undefined,
                        original_list = msgpack_list_options()}

msgpack_term() = [msgpack_term()]
                       | msgpack_map()
                       | integer()
                       | float()
                       | binary()
