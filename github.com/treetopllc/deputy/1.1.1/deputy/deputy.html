rule() = required
               | {convert, convert()}
               | {regexp, binary()}
               | {in, list()}
               | {'<', number()}
               | {'>', number()}
               | {'<=', number()}
               | {'>=', number()}
               | {min_size, integer() >= 1}
               | {max_size, integer() >= 1}
               | {size, integer() >= 1}
               | {func, {atom(), atom()} | rule_fun()}

rule_fun() = fun((term()) -> ok | {ok, term()} | error | stop)

rules() = [{rule(), binary()}]

value() = binary() | float() | integer() | boolean()
