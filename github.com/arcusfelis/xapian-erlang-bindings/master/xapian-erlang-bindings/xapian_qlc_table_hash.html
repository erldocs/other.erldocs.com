Types

gb_tree2(X, X) = gb_tree()

hr_error() = elem_not_found

hr_key() = qlc_reference() | table_hash()

hr_read_result() = {ok, {qlc_reference(), table_hash()}}
                         | {error, hr_error()}

hr_store() =
            #qlc_table_hash_register{hash_to_ref =
                                         gb_tree2(table_hash(),
                                                  qlc_reference()),
                                     ref_to_hash =
                                         gb_tree2(qlc_reference(),
                                                  table_hash())}

hr_write_result() = {ok,
                             {hr_store(),
                              qlc_reference(),
                              table_hash()}}
                          | {error, hr_error()}

qlc_reference() = reference()

table_hash() = integer() >= 0
