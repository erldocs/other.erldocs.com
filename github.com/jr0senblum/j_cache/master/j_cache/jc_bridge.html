jc_bridge

Receives requests in the form of {CallerPid::pid(), Message::tuple()}, calls the appropriate jc function and returns the result to the caller.

Receives requests in the form of {CallerPid::pid(), Message::tuple()}, calls the appropriate jc function and returns the result to the caller. Primarily used to communicate with the Java node (via jinterface).

Nearly always spawns a new process to handle the actual request.

jc_bridge currently responds to the following messages

{From, {clear, Map}} -> {ok}

{From, {contains, Map, Key}} -> {true} | {false}

{From, {evict, Map, Key}} -> {ok}

{From, {evict_match, Map, Criteria}}, State) -> {ok}

{From, {get, Map, Key}} -> {ok, V} | {miss}

{From, {get_all, Map, Keys}} -> {ok, [V1,..., Vn]} | {miss}

{From, {key_set, Map}} -> {ok, [K1,..., Kn]

{From, {put, Map, Key, Value}} -> {ok, {key, Key}}

{From, {put, Map, Key, Value, TTL}} -> {ok, {key, Key}}

{From, {put_all, Map, List}} -> {ok, N}, N is no. inserted

{From, {put, Map, List, TTL}} -> {ok, N}, N is no. inserted

{From, {values, Map}} -> {ok, [V1, V2, ..., Vn]}

{From, {values_match, Map, Criteria}} -> {ok, [{K1,V1}, {K2,V2},...]}

{From, {size}} -> {size,[ {schema,{records,6},{bytes,8632}}, {key_to_value,{records,2},{bytes,1112}}, {ttl,{records,0},{bytes,2440}}, {stats,{records,1},{bytes,2592}}, {ps_sub,{records,0},{bytes,2440}}, {ps_client,{records,0},{bytes,2440}} ] }

{From, {nodes}} -> {configured, [N1, N2,...]}

{From, {topic_subscribe, Topic, Value}} -> ok | {error, badarg}, client gets: {Topic, Value}

{From, {map_subscribe, Map, Key|any, write|delete|any}} -> ok| {error, badarg}, client gets: {map, key, write|delete}

{From, {node_topic_sub}} -> ok | {error, badarg}, client gets: {jc_node_events,{nodedown, Node}}

{From, {topic_unsubscribe, Topic, Value}} -> ok| {error, badarg},

{From, {map_unsubscribe, Map, Key, Ops}} -> ok| {error, badarg},

{From, {topic_event, Topic, Value}} -> no_return()
