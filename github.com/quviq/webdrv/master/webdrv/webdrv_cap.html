Types

capability() = #capability{} | null | jsonobj()

json() = jsonobj()
               | jsonlist()
               | jsonnum()
               | jsonstr()
               | true
               | false
               | null

jsonkey() = string()

jsonlist() = [json()]

jsonnum() = integer() | float()

jsonobj() = {obj, [{jsonkey(), json()}]}

jsonstr() = binary()

Functions

default() -> #capability{}

Return the default capability record.

default_chrome() -> #capability{}

Return the default capability with browser set to chrome.

default_htmlunit() -> #capability{}

Return the default capability with browser set to htmlunit.

default_firefox() -> #capability{}

Return the default capability with browser set to firefox.

default_safari(Device::string()) -> #capability{}

Return the default capability with browser set to firefox.

to_json(C::capability()) -> jsonobj() | null

Convert a capability (possibly null) to JSON format. Function is idempotent, i.e. converting an already converted object is fine.
