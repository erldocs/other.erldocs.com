pool_error_reasons() = {busy, allocate_request}
                             | {timeout,
                                allocate_request | postgresql_pool}

postgresql_error_reasons() = rpc_error_reasons()
                                   | pool_error_reasons()
                                   | upstream_error_reasons()

public_error_reasons() = {query_error,
                                  known_errors_codes() |
                                  {unknown_errors_codes(),
                                   nonempty_string()}}
                               | {busy | timeout | closed | nodedown,
                                  atom()}
                               | term()

rpc_error_reasons() = {nodedown, rpc}
                            | {timeout, rpc}
                            | {term(), rpc}

unknown_errors_codes() = binary()

upstream_error_reasons() = {busy, pgsql_pool}
                                 | {timeout, pgsql_pool | pgsql}
                                 | {closed, pgsql | pgsql_transaction}
                                 | {#error{} | term(), pgsql}
