                       id = undefined | binary(),
                       statement =
                           [#statement{sid = undefined | binary(),
                                       effect = allow | deny,
                                       principal = principal(),

                                                  s3_bucket_action()]

                                                      s3_bucket_action()]
                                                   | '*',
                                       resource = [arn()] | '*',
                                       condition_block =
                                           [condition_pair()]}],
                       creation_time = erlang:timestamp()}
