ACL utility functions

Types

acl() = acl1() | acl2()

acl1() =
            #acl_v1{owner = acl_owner(),
                    grants = [acl_grant()],
                    creation_time = erlang:timestamp()}

acl2() =
            #acl_v2{owner = acl_owner(),
                    grants = [acl_grant()],
                    creation_time = erlang:timestamp()}

acl_grant() = {{string(), string()}, acl_perms()}
                    | {atom(), acl_perms()}

acl_owner() = {string(), string()}
                    | {string(), string(), string()}

acl_perm() = 'READ'
                   | 'WRITE'
                   | 'READ_ACP'
                   | 'WRITE_ACP'
                   | 'FULL_CONTROL'

acl_perms() = [acl_perm()]
