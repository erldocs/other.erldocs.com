event_dispatcher(Client::fsmref()) -> pid()

Returns the event dispatcher associated to this process

send_event(Client::fsmref(), Event::client_event()) -> ok

Sends an client event to the server. The event may be one of the following ones (each one represented with a corresponding record):

client_connected

listener_disconnected

client_disconnected

set_pixel_format

set_encodings

update_request

key

pointer

client_cut_text

For a description of the events, check the RFB Protocol Definition

set_pixel_format(Client::fsmref(), PixelFormat::#pixel_format{}) -> ok

Equivalent to send_event(Client, #set_pixel_format{pixel_format = PixelFormat}).

set_encodings(Client::fsmref(), Encodings::[{integer(), atom()}]) -> ok

Equivalent to send_event(Client, #set_encodings{encodings = Encodings}).

update_request(Client::fsmref(), Incremental::boolean(), Box::#box{}) -> ok

Equivalent to send_event(Client, #update_request{incremental = Incremental, box = Box}).

key(Server::fsmref(), Down::boolean(), Code::integer()) -> ok

Equivalent to send_event(Server, #key{down = Down, code = Code}).

pointer(Server::fsmref(), ButtonMask::byte(), X::integer(), Y::integer()) -> ok

Equivalent to send_event(Server, #pointer{button_mask = ButtonMask, x = X, y = Y}).

client_cut_text(Server::fsmref(), Text::binary()) -> ok

Equivalent to send_event(Server, #client_cut_text{text = Text}).

client_disconnected(Server::fsmref(), Reason::term()) -> ok

Equivalent to send_event(Server, #client_disconnected{reason = Reason}).
