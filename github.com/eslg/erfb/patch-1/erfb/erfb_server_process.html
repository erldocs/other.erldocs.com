event_dispatcher(Server::fsmref()) -> pid()

Returns the event dispatcher associated to this process

send_event(Server::fsmref(), Event::server_event()) -> ok

Sends an server event to the client. The event may be one of the following ones (each one represented with a corresponding record):

server_connected

server_disconnected

set_colour_map_entries

update

bell

server_cut_text

For a description of the events, check the RFB Protocol Definition

update(Server::fsmref(), Rectangles::[#rectangle{}]) -> ok

Equivalent to send_event(Server, #rfbupdate{rectangles = Rectangles}).

set_colour_map_entries(Server::fsmref(), FirstColour::integer(), Colours::[#colour{}]) -> ok

Equivalent to send_event(Server, #set_colour_map_entries{first_colour = FirstColour, colours = Colours}).

bell(Server::fsmref()) -> ok

Equivalent to send_event(Server, #bell{}).

server_cut_text(Server::fsmref(), Text::binary()) -> ok

Equivalent to send_event(Server, #server_cut_text{text = Text}).

server_disconnected(Server::fsmref(), Reason::term()) -> ok

Equivalent to send_event(Server, #server_disconnected{reason = Reason}).
